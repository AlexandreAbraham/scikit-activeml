Expected Average Precision¶

 9 import numpy as np
10 from matplotlib import pyplot as plt, animation
11 from sklearn.datasets import make_blobs
12 from skactiveml.utils import MISSING_LABEL, unlabeled_indices, labeled_indices
13 from skactiveml.visualization import plot_utility, plot_decision_boundary
14 from skactiveml.pool import UncertaintySampling
15 from skactiveml.classifier import PWC
16
17 random_state = np.random.RandomState(0)
18
19 # Build a dataset.
20 X, y_true = make_blobs(n_samples=10, n_features=2,
21                   centers=[[0, 1], [-3, .5], [-1, -1], [2, 1], [1, -.5]],
22                   cluster_std=.7, random_state=random_state)
23 y_true = y_true % 2
24 y = np.full(shape=y_true.shape, fill_value=MISSING_LABEL)
25
26 # Initialise the classifier.
27 clf = PWC(classes=[0, 1], random_state=random_state)
28 # Initialise the query strategy.
29 qs = UncertaintySampling(method='expected_average_precision', random_state=random_state)
30
31 # Preparation for plotting.
32 fig, ax = plt.subplots()
33 # bound = [[min(X[:, 0]), max(X[:, 0])], [min(X[:, 1]), max(X[:, 1])]]
34 feature_bound = [[min(X[:, 0]), min(X[:, 1])], [max(X[:, 0]), max(X[:, 1])]]
35 artists = []
36
37 # The active learning cycle:
38 n_cycles = 1
39 for c in range(n_cycles):
40     # Fit the classifier.
41     clf.fit(X, y)
42
43     # Set X_cand to the unlabeled instances.
44     unlbld_idx = unlabeled_indices(y)
45     X_cand = X[unlbld_idx]
46     X_labeled = X[labeled_indices(y)]
47
48     # Query the next instance/s.
49     query_params = {"clf": clf, "X": X, "y": y}
50     query_idx = unlbld_idx[qs.query(X_cand, **query_params)]
51
52     # Plot the labeled data.
53     coll_old = list(ax.collections)
54     title = ax.text(
55         0.5, 1.05, f"Decision boundry after acquring {c} labels",
56         size=plt.rcParams["axes.titlesize"], ha="center",
57         transform=ax.transAxes
58     )
59     ax = plot_utility(qs, {'clf': clf, 'X': X, 'y': y}, X_cand=X, feature_bound=feature_bound, ax=ax)
60     ax.scatter(X[:, 0], X[:, 1], c=y_true, cmap="coolwarm", marker=".", zorder=2)
61     ax.scatter(X_labeled[:, 0], X_labeled[:, 1], c="grey", alpha=.8, marker=".", s=300)
62     ax = plot_decision_boundary(clf, feature_bound, ax=ax)
63
64     coll_new = list(ax.collections)
65     coll_new.append(title)
66     artists.append([x for x in coll_new if (x not in coll_old)])
67
68     # Label the queried instances.
69     y[query_idx] = y_true[query_idx]
70
71 ani = animation.ArtistAnimation(fig, artists, interval=1000, blit=True)
