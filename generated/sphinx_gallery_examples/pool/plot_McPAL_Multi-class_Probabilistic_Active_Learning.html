Multi-class Probabilistic Active Learning¶

 9 import numpy as np
10 from matplotlib import pyplot as plt, animation
11 from sklearn.datasets import make_blobs
12 from skactiveml.utils import MISSING_LABEL, unlabeled_indices, labeled_indices
13 from skactiveml.visualization import plot_utility, plot_decision_boundary
14 from skactiveml.pool import McPAL
15 from skactiveml.classifier import PWC
16
17 random_state = np.random.RandomState(0)
18
19 # Build a dataset.
20 X, y_true = make_blobs(n_samples=10, n_features=2,
21                   centers=[[0, 1], [-3, .5], [-1, -1], [2, 1], [1, -.5]],
22                   cluster_std=.7, random_state=random_state)
23 y_true = y_true % 2
24 y = np.full(shape=y_true.shape, fill_value=MISSING_LABEL)
25
26 # Initialise the classifier.
27 clf = PWC(classes=[0, 1], random_state=random_state)
28 # Initialise the query strategy.
29 qs = McPAL(random_state=random_state)
30 dens_est = clf.fit(X, np.zeros(len(X)))
31 dens = clf.predict_freq(X)[:,0]
32
33 # Preparation for plotting.
34 fig, ax = plt.subplots()
35 # bound = [[min(X[:, 0]), max(X[:, 0])], [min(X[:, 1]), max(X[:, 1])]]
36 feature_bound = [[min(X[:, 0]), min(X[:, 1])], [max(X[:, 0]), max(X[:, 1])]]
37 artists = []
38
39 # The active learning cycle:
40 n_cycles = 1
41 for c in range(n_cycles):
42     # Fit the classifier.
43     clf.fit(X, y)
44
45     # Set X_cand to the unlabeled instances.
46     unlbld_idx = unlabeled_indices(y)
47     X_cand = X[unlbld_idx]
48     X_labeled = X[labeled_indices(y)]
49
50     # Query the next instance/s.
51     query_params = {"clf": clf, "utility_weight": dens[unlbld_idx], "X": X, "y": y}
52     query_idx = unlbld_idx[qs.query(X_cand, **query_params)]
53
54     # Plot the labeled data.
55     coll_old = list(ax.collections)
56     title = ax.text(
57         0.5, 1.05, f"Decision boundry after acquring {c} labels",
58         size=plt.rcParams["axes.titlesize"], ha="center",
59         transform=ax.transAxes
60     )
61     ax = plot_utility(qs, {'clf': clf, 'X': X, 'y': y, 'utility_weight': dens}, X_cand=X, feature_bound=feature_bound, ax=ax)
62     ax.scatter(X[:, 0], X[:, 1], c=y_true, cmap="coolwarm", marker=".", zorder=2)
63     ax.scatter(X_labeled[:, 0], X_labeled[:, 1], c="grey", alpha=.8, marker=".", s=300)
64     ax = plot_decision_boundary(clf, feature_bound, ax=ax)
65
66     coll_new = list(ax.collections)
67     coll_new.append(title)
68     artists.append([x for x in coll_new if (x not in coll_old)])
69
70     # Label the queried instances.
71     y[query_idx] = y_true[query_idx]
72
73 ani = animation.ArtistAnimation(fig, artists, interval=1000, blit=True)
